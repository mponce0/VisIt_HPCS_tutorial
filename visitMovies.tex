\subsubsection{Annotations}
\begin{frame}
\frametitle{Annotations}
\begin{columns}
\begin{column}{5cm}
\begin{block}{Annotations}
\begin{itemize}
	\item objects in the viz-window
		that convey infromation about the plots
	\item make clear what is being visualized and make
		the visualization appear more polished
\end{itemize}
\end{block}
\end{column}
\begin{column}{5cm}
\begin{block}{Types of Annotations}
\begin{itemize}
        \item Database name
        \item User name
	\item plot legends
	\item plot axes and labels (2d \& 3d)
	\item 3d triad
	\item 2d, 3d text
	\item time slider
	\item images
	\item line and arrows
\end{itemize}
\end{block}
\end{column}
\end{columns}
\end{frame}
